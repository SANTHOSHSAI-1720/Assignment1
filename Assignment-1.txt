5.to swap two numbers with the help of a third variable:
package javaproblems;
import java.util.*;

public class multiply {
    public static void main(String[] args) {
        int num1=2;
        int num2=3;
        System.out.println("The first num before swap:" + num1);
        System.out.println("The second num before swap:" + num2);
        int temp;
        temp=num1;
        num1=num2;
        num2=temp;
        System.out.println("The first num after swap:" + num1);
        System.out.println("The second num after swap:" + num2);
    }
}
